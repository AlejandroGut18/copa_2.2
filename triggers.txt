CREATE TRIGGER grupos_por_defecto_torneo
AFTER INSERT ON torneos
FOR EACH ROW
BEGIN
  -- Grupos masculinos
  INSERT INTO grupos (nombre, status_id, torneo_id, genero) VALUES
  ('A', 1, NEW.id, 'M'),
  ('B', 1, NEW.id, 'M'),
  ('C', 1, NEW.id, 'M'),
  ('D', 1, NEW.id, 'M');

  -- Grupos femeninos
  INSERT INTO grupos (nombre, status_id, torneo_id, genero) VALUES
  ('A', 1, NEW.id, 'F'),
  ('B', 1, NEW.id, 'F'),
  ('C', 1, NEW.id, 'F'),
  ('D', 1, NEW.id, 'F');
END;

CREATE TABLE grupo_equipos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    grupo_id INT NOT NULL,
    equipo_id INT NOT NULL,
    torneo_id INT NOT NULL, -- Redundancia controlada para validaci√≥n
    UNIQUE KEY unique_equipo_torneo (equipo_id, torneo_id),
    FOREIGN KEY (grupo_id) REFERENCES grupos(id),
    FOREIGN KEY (equipo_id) REFERENCES equipos(id),
    FOREIGN KEY (torneo_id) REFERENCES torneos(id)
);
